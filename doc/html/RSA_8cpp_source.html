232 /* Performs preprocessing based on the arc lengths. An arc (u,v) can only be part of a solution if the distance from demand source to u, plus the distance from v to demand target plus the arc length is less than or equal to the demand's maximum length. */
233 bool RSA::lengthPreprocessing(){
234  int totalNb = 0;
235  for (int d = 0; d < getNbDemandsToBeRouted(); d++){
236  //displayGraph(d);
237  int nb = 0;
238  ListDigraph::ArcIt previousArc(*vecGraph[d]);
239  ListDigraph::ArcIt a(*vecGraph[d]);
240  ListDigraph::ArcIt currentArc(*vecGraph[d], a);
241  while (a != INVALID){
242  currentArc = a;
243  ListDigraph::ArcIt nextArc(*vecGraph[0], ++currentArc);
244  currentArc = a;
245  int slice = getArcSlice(a, d);
246  ListDigraph::Node source = getNode(d, getToBeRouted_k(d).getSource(), slice);
247  ListDigraph::Node target = getNode(d, getToBeRouted_k(d).getTarget(), slice);
248  if (source != INVALID && target != INVALID){
249  if (shortestDistance(d, source, a, target) >= getToBeRouted_k(d).getMaxLength() + DBL_EPSILON){
250  (*vecGraph[d]).erase(a);
251  nb++;
252  }
253  }
254  else{
255  nb += eraseAllArcsFromSlice(d, slice);
256  }
257  a = nextArc;
258  }
259  std::cout << "> Number of erased arcs due to length in graph #" << d << ": " << nb << std::endl;
260  totalNb += nb;
261  }
262  if (totalNb >= 1){
263  return true;
264  }
265  return false;
266 }

268 /* Returns the distance of the shortest path from source to target passing through arc a. */
269 double RSA::shortestDistance(int d, ListDigraph::Node &s, ListDigraph::Arc &a, ListDigraph::Node &t){
270  double distance = 0.0;
271  Dijkstra< ListDigraph, ListDigraph::ArcMap<double> > s_u_path((*vecGraph[d]), (*vecArcLength[d]));
272 
273  s_u_path.run(s,(*vecGraph[d]).source(a));
274  if (s_u_path.reached((*vecGraph[d]).source(a))){
275  distance += s_u_path.dist((*vecGraph[d]).source(a));
276  }
277  else{
278  return DBL_MAX;
279  }
280 
281  distance += getArcLength(a, d);
282 
283  Dijkstra< ListDigraph, ListDigraph::ArcMap<double> > v_t_path((*vecGraph[d]), (*vecArcLength[d]));
284  v_t_path.run((*vecGraph[d]).target(a), t);
285  if (v_t_path.reached(t)){
286  distance += v_t_path.dist(t);
287  }
288  else{
289  return DBL_MAX;
290  }
291 
292  return distance;
293 }
