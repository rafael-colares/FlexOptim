323 double RSA::shortestDistance(int d, ListDigraph::Node &s, ListDigraph::Arc &a, ListDigraph::Node &t){
324  double distance = 0.0;
325  Dijkstra< ListDigraph, ListDigraph::ArcMap<double> > s_u_path((*vecGraph[d]), (*vecArcLength[d]));
326 
327  s_u_path.run(s,(*vecGraph[d]).source(a));
328  if (s_u_path.reached((*vecGraph[d]).source(a))){
329  distance += s_u_path.dist((*vecGraph[d]).source(a));
330  }
331  else{
332  return DBL_MAX;
333  }
334 
335  distance += getArcLength(a, d);
336 
337  Dijkstra< ListDigraph, ListDigraph::ArcMap<double> > v_t_path((*vecGraph[d]), (*vecArcLength[d]));
338  v_t_path.run((*vecGraph[d]).target(a), t);
339  if (v_t_path.reached(t)){
340  distance += v_t_path.dist(t);
341  }
342  else{
343  return DBL_MAX;
344  }
